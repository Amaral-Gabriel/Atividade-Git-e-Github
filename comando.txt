# git version
    Verifica a versão intalada na maquina

# git config
    Altera as informações do usuário no commit

# --global
    Altera as informações do usuário em todo o repositório

# git init
    inicializa um novo repositório

# git status   
    Mostra a branch atual e possívies commits

# git add
    Adiciona arquivos ao staging

# git rm
    Remove arquivos do staging

# --cached
    Remove o arquivo do staging sem apagar do repositório

# -r
    Remove diretórios do staging

# git commit 
    Salva as mudanças feitas no staging, no repositório local

# -m
    Adiciona uma mensagem ao commit

# git remote
    Referencia o repositório local com o remoto

# git push
    Manda as modificações feitas no repositório local para o remoto

# git clone
    Clona um repositório remoto via HTTPS ou SSH Ref

# git pull
    Atualiza um repositório local com um remoto

# git checkout
    Refaz as mudanças antes de serem adicionadas no staging
